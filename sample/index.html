CSS

// IE6 .ex-child .menu-parent { background: url("images/i_ar.gif") no-repeat left top; padding-left: 20px; } .ex-child .menu-child { padding: 5px 0; margin: 10px 0 0 10px; border-bottom: 1px dashed #333; } // other .ex-child > ul > li { background: url("images/i_ar.gif") no-repeat left top; padding-left: 20px; } .ex-child > ul > li > ul > li { padding: 5px 0; margin: 10px 0 0 10px; border-bottom: 1px dashed #333; }
